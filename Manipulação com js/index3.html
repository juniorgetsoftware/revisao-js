Objetos em JS

Declarando o objeto
var objeto = {};
Índices
var objeto = {
placa: 'ADD-443', modelo: 'Sedã', marca: 'Hyundai'
}; console.log("Adicionando índices (atributos): ", objeto);
Imprimindo atributos
console.log(objeto.placa);
console.log(objeto.marca);
Atributo de objeto
var objeto = {

placa: 'ADD-443',
modelo: 'Sedã',
marca: 'Hyundai',
opcionais: {

cambio: 'automático',
combustivel: 'flex'
}

};
console.log("Atributo do tipo object: ", objeto);
console.log(objeto.opcionais.cambio);
Atributo composto
var objeto = {

placa: 'ADD-443',
modelo: 'Sedã',
marca: 'Hyundai',
opcionais: {

cambio: 'automático',
combustivel: 'flex'
},
"possui-ar-condicionado" : true

};
console.log(objeto["possui-ar-condicionado"]);
